                        C_baggage, Cb_set),

(** Conversion function from Dis_passenger to C_person *)

let disToCp (p : Dis_passenger) : C_person = !depToCp((DisToDep!to_super(p)));

(** conversion function from Dep_passenger to C_person *)

let depToCp (p : Dep_passenger) : C_person =
  CpaToCp!to_super(DepToCpa!to_super(p));
